What is D3?

D3 is a JavaScript library for manipulating documents based on data.

How does it do it?

Binds data arbitrarily to DOM

Display that data with HTML/CSS/SVG

Transform and transition based on data/interaction

JS contd...



  var Vis = function(target, data) {                                                           
    //init                                                                                              
    this.data = data;                                                                                   
    this.target = target;
                                                                                                        
    //abstract out into d3_container                                                                    
    this.Container =                                                                                 
      d3.select("#"+this.target)                                                                      
        .append("svg")                                                                                  
        .attr("class", "vis")                                                                  
        .attr("width", 1080)                                                                            
    this.Nodes = this.container
      .selectAll("svg:g")
      .data(this.data);
  };

More JS...


  Vis.prototype =  {

      this.Container                                                                                 
        .transition()                                                                                   
        .duration(1000)                                                                                 
        .attr("width", 1200).attr("height", this.h*110);                                                

      this.Nodes
        .enter()
        .append("circle")
        .attr("class", "node")                                                                          
        .attr("id", function(d) { return d.key})                                                        
        .attr("width", 1080); 
    }
