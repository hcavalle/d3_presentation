What is D3?

D3 is a JavaScript library for manipulating documents based on data.

How does it do it?

Bind (and manipulte) data arbitrarily to DOM

Display that data with HTML/CSS/SVG

Transform and transition based on data/interaction

    this.target = target;

    this.Container =                                                                                 
      d3.select("#"+this.target)

Transitions

Animate and transform


  Vis.prototype =  {                                                                            
    go: function go() {                                                                                 
      this.Container                                                                                 
        .transition()                                                                                   
        .duration(1000)                                                                                 
        .attr("width", 1200).attr("height", this.h*110);                                                

      this.Nodes
        .enter()
        .append("circle")
        .attr("class", "node")                                                                          
        .attr("id", function(d) { return d.key})

    }

Load data

Load the data and 'go' 


 d3.json("path/to/jsonfile", function(error, json) {                                                
    if (error) return console.warn(error);                                                              
    data = json;                                                                                        
    var vis = new Vis('body', data);                                                   
    vis.go();
  });
